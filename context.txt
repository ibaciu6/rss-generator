7. Code Examples
Scraper

scraper/scrape.py

import asyncio
from playwright.async_api import async_playwright
from lxml import html

URL = "https://sitefilme.com/"

async def scrape():

    async with async_playwright() as p:

        browser = await p.chromium.launch(headless=True)

        page = await browser.new_page()

        await page.goto(URL, timeout=60000)

        await page.wait_for_timeout(5000)

        content = await page.content()

        tree = html.fromstring(content)

        movies = []

        for item in tree.xpath("//article"):

            title = item.xpath(".//h2/a/text()")

            link = item.xpath(".//h2/a/@href")

            if title and link:

                movies.append({
                    "title": title[0],
                    "link": link[0]
                })

        await browser.close()

        return movies


if __name__ == "__main__":

    movies = asyncio.run(scrape())

    for m in movies:
        print(m)

from feedgen.feed import FeedGenerator
import json

def generate(items):

    fg = FeedGenerator()

    fg.title("SiteFilme New Movies")
    fg.link(href="https://sitefilme.com/")
    fg.description("Latest movies from sitefilme")

    for item in items:

        fe = fg.add_entry()

        fe.title(item["title"])

        fe.link(href=item["link"])

        fe.guid(item["link"])

        fe.description(item["title"])

    fg.rss_file("feeds/sitefilme.xml")


if __name__ == "__main__":

    with open("data/cache.json") as f:
        items = json.load(f)

    generate(items)

Generic scraper engine
import httpx
import cloudscraper
from playwright.sync_api import sync_playwright

def fetch(url, method="auto"):

    if method == "http":

        r = httpx.get(url, timeout=30)

        return r.text

    if method == "cloudscraper":

        scraper = cloudscraper.create_scraper()

        r = scraper.get(url)

        return r.text

    if method == "playwright":

        with sync_playwright() as p:

            browser = p.chromium.launch()

            page = browser.new_page()

            page.goto(url)

            page.wait_for_timeout(4000)

            content = page.content()

            browser.close()

            return content

from lxml import html

def parse(content, config):

    tree = html.fromstring(content)

    items = []

    for el in tree.xpath(config["item_selector"]):

        title = el.xpath(config["title_selector"])

        link = el.xpath(config["link_selector"])

        if title and link:

            items.append({
                "title": title[0],
                "link": link[0]
            })

    return items

scraper/fetcher.py

import httpx
import cloudscraper
from playwright.sync_api import sync_playwright


def fetch_http(url):

    r = httpx.get(url, timeout=30)

    return r.text


def fetch_cloudscraper(url):

    scraper = cloudscraper.create_scraper()

    r = scraper.get(url)

    return r.text


def fetch_playwright(url):

    with sync_playwright() as p:

import httpx
import cloudscraper
from playwright.sync_api import sync_playwright


def fetch_http(url):

    r = httpx.get(url, timeout=30)

    return r.text


def fetch_cloudscraper(url):

    scraper = cloudscraper.create_scraper()

    r = scraper.get(url)

    return r.text


def fetch_playwright(url):

    with sync_playwright() as p:

        browser = p.chromium.launch(headless=True)

        page = browser.new_page()

        page.goto(url, timeout=60000)

        page.wait_for_timeout(3000)

        html = page.content()

        browser.close()

        return html

10. Deduplication
core/dedup.py
import json
import os

CACHE_FILE = "data/cache.json"


def load_cache():

    if not os.path.exists(CACHE_FILE):
        return set()

    with open(CACHE_FILE) as f:
        return set(json.load(f))


def save_cache(items):

    with open(CACHE_FILE, "w") as f:
        json.dump(list(items), f)
